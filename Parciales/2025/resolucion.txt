SUBRUTINA CARGA(E/S ENTERO P_CODIGO[50], E/S CARACTER P_NOMBRE[50][26]
E/S REAL P_VALOR_H[50],E/S REAL HS_POR_PROYECTO[50] , E/S REAL HS_POR_MES[11])
//VAR
ARCHIVO: ARC
CARACTER: CLIENTE[26]
ENTERO: ID_EMP, DIA, MES, COD, INDICE, I
REAL: HORAS
//INICIO
ARC <- ABRIR("Proyectos.txtâ€‹","R")
I<-1
MIENTRAS(NOT EOF(ARC))
	LEER(ARC,P_CODIGO[I],P_NOMBRE[I], CLIENTE, P_VALOR_H[I])
	I <- I + 1
FIN MIENTRAS
CERRAR(ARC)

PARA I<-1,50
	HS_POR_PROYECTO[I] <- 0
FIN PARA

PARA I<-1,11
	HS_POR_MES[I]<-0
FIN PARA

ARC <- ABRIR("Horas2025.txt","R")
MIENTRAS(NOT EOF(ARC))
	LEER(ARC,ID_EMP,DIA,MES,COD,HORAS)
	INDICE <- BUSQUEDA(COD,P_CODIGO)
	HS_POR_PROYECTO[INDICE] <- HS_POR_PROYECTO[INDICE] + HORAS
	HS_POR_MES[INDICE] <- HS_POR_MES[INDICE] + HORAS
FIN MIENTRAS
CERRAR(ARC)

FIN SUBRUTINA

FUNCION BUSQUEDA(E/S ENTERO P_CODIGO[50], ENTERO COD_BUSCADO): ENTERO
//VAR
ENTERO: I
//INICIO
I <- 1
MIENTRAS(P_CODIGO[I] <> COD_BUSCADO AND I<=50)
	I <- I + 1
FIN MIENTRAS
SI(P_CODIGO[I] == COD_BUSCADO)
	DEVOLVER I
SINO
	DEVOLVER -1
FIN SI
FIN FUNCION

FUNCION MAYOR(E/S REAL HS_POR_MES[11]): ENTERO
//VAR
ENTERO: MES_MAYOR, I
//INICIO
MES_MAYOR <- 1
PARA I<-2,11
	SI(HS_POR_MES[I] > HS_POR_MES[MES_MAYOR])
		MES_MAYOR <- I
	FIN SI
FIN PARA
DEVOLVER INDICE_MAYOR
FIN FUNCION

SUBRUTINA RECAUDACION_PROY(E/S CARACTER P_NOMBRE[50][26], E/S REAL P_VALOR_H[50],
E/S REAL HS_POR_PROYECTO[50])
//VAR
ENTERO: I
//INICIO
PARA I<-1,50
	ESCRIBIR("El proyecto de nombre", P_NOMBRE[I],
	"RECAUDO":, P_VALOR_H[I]*HS_POR_PROYECTO[I])
FIN PARA
FIN SUBRUTINA

SUBRUTINA HORAS_TRABAJADAS(E/S ENTERO P_CODIGO[50], E/S REAL HS_POR_PROYECTO[50], ENTERO COD_INGRESADO)
//VAR

//INICIO
INDICE <- BUSQUEDA(P_CODIGO,COD_INGRESADO)
SI( INDICE <> -1)
	ESCRIBIR("Se trabajaron ", HS_POR_PROYECTO[INDICE], "en el proyecto seleccionado")
SINO
	ESCRIBIR("CODIGO INCORRECTO")
FIN SI
FIN SUBRUTINA
