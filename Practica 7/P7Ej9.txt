SUBRUTINA LECTURA(E/S REAL T[50][31])
//VAR
ARCHIVO: ARC
ENTERO: DIA, CIUDAD
REAL: TEMP
//INICIO
ARC <- ABRIR ("TEMPS.DAT","R")
MIENTRAS(NOT EOF(ARC))
	LEER(ARC,DIA,CIUDAD,TEMP)
	T[CIUDAD][DIA] <- TEMP
FIN MIENTRAS
CERRAR(ARC)
FIN SUBRUTINA

FUNCION CUANTOS(E: ENTERO COD, E: ENTERO T, E/S: REAL TEMP[50][31]): ENTERO
//VAR
ENTERO: I, CONT
//INICIO
CONT <-0
PARA I<1,31
	SI(TEMP[COD][I] > T)
		CONT<- CONT + 1
	FIN SI
FIN PARA
DEVOLVER(CONT)
FIN FUNCION

SUBRUTINA MAYORPORCIUDAD(E/S REAL T[50][31],E/S REAL T_MAYOR[50],E/S ENTERO DIA_MAYOR[50])
//VAR
ENTERO: I, POS_MAYOR
//INICIO
PARA J<-1,50
	POS_MAYOR<-1
	PARA I<-2,31
		SI(TEMP[J][I] > TEMP[J][POS_MAYOR])
			POS_MAYOR<- I
		FIN SI
	FIN PARA
	T_MAYOR[J]<-TEMP[J][POS_MAYOR]
	DIA_MAYOR[J]<-POS_MAYOR
FIN PARA
FIN SUBRUTINA

SUBRUTINA ORDENAYMUESTRA(E/S REAL T_MAYOR[50],E/S ENTERO DIA_MAYOR[50])
//VAR
ENTERO: NRO_CIUDAD[50],I,J,MIN
//INICIO
PARA I<-1,50
	NRO_CIUDAD[I]<-I
FIN PARA

PARA J<-50,2,-1
	MIN<- 1
	PARA I<-1,J
		SI(T_MAYOR[I] < T_MAYOR[MIN])
			MIN<-I
		FIN SI
	FIN PARA
	INTERCAMBIOREAL( T_MAYOR[J] , T_MAYOR[MIN])				//REALIZO EL INTERCAMBIO DE 
	INTERCAMBIOENTERO( DIA_MAYOR[J] , DIA_MAYOR[MIN] )		//VARIABLES A TRAVES DE UNA
	INTERCAMBIOENTERO( NRO_CIUDAD[J] , NRO_CIUDAD[MIN] )	//SUBRUTINA
FIN PARA

PARA I<-1,50
	ESCRIBIRI("LA CIUDAD NUMERO ", NRO_CIUDAD[I]," TIENE SU MAYOR TEMP ", T_MAYOR[I], "EL DIA", DIA_MAYOR[I])
FIN PARA
FIN SUBRUTINA

ALGORITMO PRINCIPAL
//VAR
REAL: T[50][31], T_MAYOR[50]
ENTERO: COD, TEMP, DIA_MAYOR[50]
//INICIO
LECTURA(T)
ESCRIBIR("INGRESE UN CODIGO DE CIUDAD")
LEER(COD)
MIENTRAS(COD<>0)
	ESCRIBIR("INGRESE UN VALOR DE TEMPERATURA")
	LEER(TEMP)
	ESCRIBIR("LA CANTIDAD DE DIAS CON TEMPERATURA MAYORES A ", TEMP, "ES", CUANTOS(COD, TEMP, T))
	ESCRIBIR("INGRESE UN CODIGO DE CIUDAD")
	LEER(COD)
FIN MIENTRAS

MAYORPORCIUDAD(T, T_MAYOR, DIA_MAYOR)
ORDENAYMUESTRA(T_MAYOR, DIA_MAYOR)

FIN ALGORITMO
