ALGORITMO PRINCIPAL
//VAR
ENTERO: KM_INGRESADO, CONT_POR_KM[500], SUMA_POR_KM[500]
REAL: PROM_PARA_KM
//INICIO
CARGA(CONT_POR_KM,SUMA_POR_KM)
CANT_VECES(CONT_POR_KM)
MAS_VECES(CONT_POR_KM)
ESCRIBIR("INGRESE UN KILOMETRO")
LEER(KM_INGRESADO)
PROM_PARA_KM <- CONSULTA(CONT_POR_KM,SUMA_POR_KM,KM_INGRESADO)
ESCRIBIR("EL PROMEDIO PARA EL KM INGRESADO ES", PROM_PARA_KM)
ORDENA(CONT_POR_KM,SUMA_POR_KM)
FIN ALGORITMO

SUBRUTINA CARGA(E/S ENTERO: CONT_POR_KM[500], E/S ENTERO: SUMA_POR_KM[500])
//VAR
ARCHIVO: ARCH
ENTERO: KM, ALT, AÑO, I
//INICIO
PARA I<-1,500
	CONT_POR_KM[I] <- 0
	SUMA_POR_KM[I] <- 0
FIN PARA

ARCH <- ABRIR("ALT_RIO.DAT","R")
MIENTRAS(NOT EOF(ARCH))
	LEER(ARCH,KM,ALT,AÑO)
	CONT_POR_KM[KM] <- CONT_POR_KM[KM] + 1
	SUMA_POR_KM[KM] <- SUMA_POR_KM[KM] + ALT
FIN MIENTRAS
CERRAR(ARCH)

FIN SUBRUTINA


SUBRUTINA CANT_VECES(E/S ENTERO: CONT_POR_KM[500])
// VAR
ENTERO: I
//INICIO
PARA I<-1,500
	ESCRIBIR("PARA EL KILOMETRO", I, "HAY", CONT_POR_KM[I],"MEDICIONES")
FIN PARA
FIN SUBRUTINA

SUBRUTINA MAS_VECES(E/S ENTERO: CONT_POR_KM[500])
//VAR

//INICIO
POS_MAX <- 1
PARA I<2,500
	SI(CONT_POR_KM[I] > CONT_POR_KM[POS_MAX])
		POS_MAX <- I
	FIN SI
FIN PARA
ESCRIBIR("EL KILOMETRO CON MAS MEDICIONES ES", POS_MAX")
FIN SUBRUTINA

FUNCION CONSULTA(E/S ENTERO: CONT_POR_KM[500], E/S ENTERO: SUMA_POR_KM[500], E: ENTERO: KM_INGRESADO): REAL
DEVOLVER SUMA_POR_KM[KM_INGRESADO] * 1.0 / CONT_POR_KM[KM_INGRESADO]
FIN FUNCION

SUBRUTINA ORDENA(E/S ENTERO: CONT_POR_KM[500], E/S ENTERO: SUMA_POR_KM[500])
//VAR
REAL: PROMEDIOS[500], AUX_R
ENTERO: I, POSICIONES[500], AUX_E
//INICIO
PARA I<-1,500
	PROMEDIOS[I] <- SUMA_POR_KM[I] * 1.0 / CONT_POR_KM[I]
	POSICIONES[I] <- I
FIN PARA

PARA J<-500,2,-1
	POS_MAX <- 1
	PARA I<1,J
		SI(PROMEDIOS[I] > PROMEDIOS[POS_MAX])
			POS_MAX <- I
		FIN SI
	FIN PARA
	AUX_R <- PROMEDIOS[POS_MAX]
	PROMEDIOS[POS_MAX] <- PROMEDIOS[J]
	PROMEDIOS[J] <- AUX

	AUX_E <- POSICIONES[POS_MAX]
	POSICIONES[POS_MAX] <- POSICIONES[J]
	POSICIONES[J] <- AUX
FIN PARA

PARA I<-1,500
	ESCRIBIR("EL KM", POSICIONES[I]," TIENE UNA ALTURA PROMEDIO DE", PROMEDIOS[I])
FIN PARA
FIN SUBRUTINA
