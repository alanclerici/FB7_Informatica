ALGORITMO P6EJ2
//VAR
	REAL V[20], VALOR[20]
	ENTERO POSICION[20], I
//INICIO
	PARA I<-1,20
		ESCRIBIR("INGRESE EL VALOR DE LA POSICION", I)
		LEER(V[I])
	FIN PARA
	INDICE <- 1
	PARA I<-1,20
		SI(V[I]<>0)
			VALOR[INDICE]<-V[I]
			POSICION[INDICE]<-I
			INDICE <- INDICE + 1
		FIN SI
	FIN PARA
	PARA I<-1,(INDICE - 1)
		ESCRIBIR("VALOR:",VALOR[I],"POSICION:",POSICION[I])
	FIN PARA
FIN ALGORITMO
